import { Axiom } from '@axiomhq/js'

const axiom = new Axiom({
  token: process.env.AXIOM_TOKEN,
})

// Query the last 24h of events
const result = await axiom.query(`
  ['production-logs']
  | where status >= 500
  | summarize count() by bin(_time, 1h)
`, { startTime: '24h' })
